# Schema Definitions

JSON schemas for input and output files used by the roadmap optimizer.

## Job JSON Schema

Input schema for individual SAS job definitions:

```json
{
  "$schema": "http://json-schema.org/draft-07/schema#",
  "type": "object",
  "required": ["job_id", "domain", "steps"],
  "properties": {
    "job_id": {
      "type": "string",
      "description": "Unique identifier for the job"
    },
    "domain": {
      "type": "string",
      "description": "Business domain this job belongs to (user-configurable, e.g., 'Claims', 'Policies', 'Entities', 'customer', 'finance')"
    },
    "steps": {
      "type": "array",
      "items": {
        "type": "object",
        "properties": {
          "step_id": {"type": "string"},
          "step_type": {"type": "string"},
          "reads_from": {"type": "array", "items": {"type": "string"}},
          "writes_to": {"type": "array", "items": {"type": "string"}},
          "transformations": {"type": "array", "items": {"type": "string"}}
        }
      }
    },
    "upstream_dependencies": {
      "type": "array",
      "items": {"type": "string"},
      "description": "Job IDs that must complete before this job"
    },
    "downstream_dependencies": {
      "type": "array",
      "items": {"type": "string"},
      "description": "Job IDs that depend on this job"
    },
    "source_systems": {
      "type": "array",
      "items": {"type": "string"},
      "description": "External systems this job reads from"
    },
    "volume_indicators": {
      "type": "object",
      "properties": {
        "daily_records": {"type": "number"},
        "monthly_records": {"type": "number"},
        "data_size_gb": {"type": "number"},
        "size_factor": {"type": "number", "minimum": 0.1, "maximum": 10.0}
      }
    },
    "complexity_indicators": {
      "type": "object",
      "properties": {
        "transformation_complexity": {"type": "number"},
        "business_logic_complexity": {"type": "number"},
        "data_quality_issues": {"type": "array", "items": {"type": "string"}}
      }
    },
    "business_context": {
      "type": "object",
      "properties": {
        "criticality": {"type": "string", "enum": ["low", "medium", "high", "critical"]},
        "regulatory_requirements": {"type": "array", "items": {"type": "string"}},
        "stakeholder_groups": {"type": "array", "items": {"type": "string"}},
        "change_frequency": {"type": "string", "enum": ["stable", "occasional", "frequent"]}
      }
    }
  }
}
```

## Package Output Schema

Schema for migration package definitions generated by aggregation:

```json
{
  "$schema": "http://json-schema.org/draft-07/schema#",
  "type": "array",
  "items": {
    "type": "object",
    "required": ["package_id", "name", "domain", "job_ids", "total_effort_days"],
    "properties": {
      "package_id": {
        "type": "string",
        "description": "Unique identifier for the migration package"
      },
      "name": {
        "type": "string",
        "description": "Human-readable package name"
      },
      "domain": {
        "type": "string",
        "description": "Primary business domain"
      },
      "job_ids": {
        "type": "array",
        "items": {"type": "string"},
        "description": "SAS job IDs included in this package"
      },
      "job_count": {
        "type": "integer",
        "minimum": 1,
        "description": "Number of jobs in package"
      },
      "total_effort_days": {
        "type": "number",
        "minimum": 0,
        "description": "Estimated total effort in person-days"
      },
      "complexity_score": {
        "type": "number",
        "minimum": 1.0,
        "description": "Average complexity score (1-10 scale)"
      },
      "upstream_packages": {
        "type": "array",
        "items": {"type": "string"},
        "description": "Job IDs this package depends on (external to package)"
      },
      "downstream_packages": {
        "type": "array",
        "items": {"type": "string"},
        "description": "Job IDs that depend on this package (external to package)"
      },
      "upstream_count": {
        "type": "integer",
        "description": "Number of upstream dependencies"
      },
      "downstream_count": {
        "type": "integer",
        "description": "Number of downstream dependencies"
      },
      "centrality_score": {
        "type": "number",
        "minimum": 0,
        "maximum": 1,
        "description": "Betweenness centrality in dependency graph"
      },
      "business_value": {
        "type": "number",
        "minimum": 0,
        "maximum": 1,
        "description": "Normalized business value score"
      },
      "risk_score": {
        "type": "number",
        "minimum": 0,
        "maximum": 1,
        "description": "Migration risk assessment score"
      }
    }
  }
}
```

## Configuration Schema

Schema for optimization configuration files:

```json
{
  "$schema": "http://json-schema.org/draft-07/schema#",
  "type": "object",
  "required": ["migration_horizon_months", "odl_completion_month", "team_capacity"],
  "properties": {
    "migration_horizon_months": {
      "type": "integer",
      "minimum": 12,
      "maximum": 120,
      "description": "Total migration timeline in months"
    },
    "odl_completion_month": {
      "type": "integer",
      "minimum": 1,
      "description": "Month when ODL project completes"
    },
    "team_capacity": {
      "type": "integer",
      "minimum": 1,
      "description": "Total number of team members available"
    },
    "domain_constraints": {
      "type": "object",
      "patternProperties": {
        "^[a-z_]+$": {
          "type": "object",
          "properties": {
            "earliest_start": {"type": "integer", "minimum": 0},
            "priority_weight": {"type": "number", "minimum": 0, "maximum": 1}
          }
        }
      }
    },
    "mode_parameters": {
      "type": "object",
      "properties": {
        "build_to_legacy": {
          "$ref": "#/definitions/modeParams"
        },
        "bridge_to_model": {
          "$ref": "#/definitions/modeParams"
        },
        "strategic": {
          "$ref": "#/definitions/modeParams"
        }
      }
    },
    "system_deadlines": {
      "type": "object",
      "description": "Source system decommissioning deadlines",
      "patternProperties": {
        "^[A-Za-z_]+$": {
          "type": "object",
          "properties": {
            "decommission": {"type": "string", "description": "End-of-life date (ISO 8601)"},
            "migration_deadline": {"type": "string", "description": "Latest completion date for dependent clusters"},
            "affected_domain": {"type": "string", "description": "Primary domain affected"}
          }
        }
      }
    },
    "role_rates": {
      "type": "object",
      "description": "Hourly rates per role for cost calculations",
      "properties": {
        "architect": {"type": "number", "description": "EUR/hour"},
        "sr_data_engineer": {"type": "number"},
        "data_engineer": {"type": "number"},
        "qa_engineer": {"type": "number"},
        "business_analyst": {"type": "number"}
      }
    },
    "concurrent_limits": {
      "type": "object",
      "properties": {
        "max_strategic_parallel": {"type": "integer", "default": 2, "description": "Max Strategic clusters in parallel"},
        "max_total_parallel": {"type": "integer", "default": 4, "description": "Max total clusters in-flight"}
      }
    }
  },
  "definitions": {
    "modeParams": {
      "type": "object",
      "properties": {
        "duration_multiplier": {"type": "number", "minimum": 0.1, "maximum": 3.0},
        "tech_debt_penalty": {"type": "number", "minimum": 0},
        "resource_efficiency": {"type": "number", "minimum": 0.1, "maximum": 2.0},
        "requires_odl": {"type": "boolean"},
        "base_hours": {"type": "number", "description": "Base effort hours (e.g., 438 for Strategic, 150 for BtM)"},
        "cost_multiplier": {"type": "number", "description": "Cost scaling factor"},
        "requires_future_core": {"type": "boolean", "description": "Whether this approach requires future core availability"}
      }
    }
  }
}
```

## Optimization Result Schema

Schema for optimization results and schedules:

```json
{
  "$schema": "http://json-schema.org/draft-07/schema#",
  "type": "object",
  "required": ["scenario_name", "total_duration_months", "schedule"],
  "properties": {
    "scenario_name": {
      "type": "string",
      "description": "Name of optimization scenario"
    },
    "total_duration_months": {
      "type": "integer",
      "minimum": 1,
      "description": "Total migration duration"
    },
    "packages_by_mode": {
      "type": "object",
      "properties": {
        "build_to_legacy": {"type": "integer", "minimum": 0},
        "bridge_to_model": {"type": "integer", "minimum": 0},
        "strategic": {"type": "integer", "minimum": 0}
      }
    },
    "resource_utilization": {
      "type": "number",
      "minimum": 0,
      "maximum": 1,
      "description": "Average team utilization rate"
    },
    "technical_debt_score": {
      "type": "number",
      "minimum": 0,
      "description": "Total technical debt score"
    },
    "strategic_coverage": {
      "type": "number",
      "minimum": 0,
      "maximum": 1,
      "description": "Proportion of effort using strategic mode"
    },
    "schedule": {
      "type": "array",
      "items": {
        "type": "object",
        "required": ["package_id", "start_month", "end_month", "selected_mode"],
        "properties": {
          "package_id": {"type": "string"},
          "package_name": {"type": "string"},
          "domain": {"type": "string"},
          "start_month": {"type": "integer", "minimum": 0},
          "end_month": {"type": "integer", "minimum": 1},
          "duration_months": {"type": "integer", "minimum": 1},
          "selected_mode": {
            "type": "string",
            "enum": ["build_to_legacy", "bridge_to_model", "strategic"]
          },
          "effort_days": {"type": "number", "minimum": 0},
          "business_value": {"type": "number"},
          "assigned_team": {"type": "string"},
          "dependencies_met": {"type": "boolean"},
          "risk_factors": {
            "type": "array",
            "items": {"type": "string"}
          }
        }
      }
    },
    "objective_value": {
      "type": "number",
      "description": "Final objective function value"
    },
    "constraint_violations": {
      "type": "array",
      "items": {
        "type": "object",
        "properties": {
          "constraint_type": {"type": "string"},
          "violation_description": {"type": "string"},
          "severity": {"type": "string", "enum": ["warning", "error"]}
        }
      }
    },
    "solver_statistics": {
      "type": "object",
      "properties": {
        "solve_time_seconds": {"type": "number"},
        "optimality_gap": {"type": "number"},
        "iterations": {"type": "integer"},
        "variables": {"type": "integer"},
        "constraints": {"type": "integer"}
      }
    }
  }
}
```
